    return {
      name: document.querySelector('#productTitle')?.innerText.trim(),
      price: document.querySelector('.a-price-whole')?.innerText.trim(),
      image: document.querySelector('#landingImage')?.src,
      url: url

      price: document.querySelector('._30jeq3')?.innerText.trim(),
      image: document.querySelector('._396cs4')?.src,
      url: url
    };
  }
  
  return null;
}
